import sympy as sp

def find_derivative(expression):
    x = sp.Symbol('x')
    derivative = sp.diff(expression, x)
    return derivative

def main():
    # Get the function expression from the user
    expression_str = input("Enter a function expression in terms of x: ")

    # Convert the user input string to a SymPy expression
    expression = sp.sympify(expression_str)

    # Check if the expression is valid
    if isinstance(expression, sp.Expr):
        # Find the derivative using the provided function
        derivative = find_derivative(expression)

        # Print the derivative
        print("The derivative of", expression_str, "is", derivative)
    else:
        print("Invalid input. Please enter a valid function expression.")

if __name__ == "__main__":
    main()
