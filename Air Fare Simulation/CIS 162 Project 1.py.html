# Phase 1 & 2 | 2-1-24 4:39 PM |

#Promting user for inputs regarding their travels
print("Enter information")
destination = input("Destination: ")
day = input("Day: ")
row = int(input("Row: "))
bags = int(input("Bags: "))
premium_plus = input("Premium Plus (y/n): ")



#Checking for errors (Phase 7) | 2-3-24 10:20 PM

#Checking if user entered row is between 1-35, if entered destination
# is valid as well as the day

#Checking if row is between 1-35
if not 1 <= row <= 35:
    print("Oops! Row must be 1 - 35.")
    exit()

#checking if destination is valid
valid_airports = ["DFW", "LAX", "SEA", "MIA", "BOS"]
if destination.upper() not in valid_airports:
    print("Oops! Destination must be DFW, LAX, SEA, MIA or BOS.")
    exit()

#checking if day is valid
valid_days = ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"]
if day.capitalize() not in valid_days:
    day = "Mon"

#checking if bag is a negative number
if bags < 0:
    print("Oops! Please enter a number greater than 0.")
    exit()

#checking if row is a negative number
if row < 0:
    print("Oops! Rows must be between 0 - 35.")
    exit()

#Checking if a user leaves something empty or not
if destination.strip() == '' or day.strip() == '' or premium_plus.strip() == '':
    print("Oops! Please fill in all the required information.")
    exit()

# Designating default fares (Phase 2)

# Fares for Dallas Fort Worth
if destination == "DFW":
    if day == "Mon":
        fare = 213
    elif day == "Tue":
        fare = 198
    elif day == "Wed":
        fare = 198
    elif day == "Thu":
        fare = 213
    elif day == "Fri":
        fare = 399
    elif day == "Sat":
        fare = 213
    else:
        fare = 198

# Fares for Los Angeles International Airport
elif destination == "LAX":
    if day == "Mon":
        fare = 334
    elif day == "Tue":
        fare = 314
    elif day == "Wed":
        fare = 314
    elif day == "Thu":
        fare = 334
    elif day == "Fri":
        fare = 429
    elif day == "Sat":
        fare = 334
    else:
        fare = 314

# Fares for Miami International Airport
elif destination == "MIA":
    if day == "Mon":
        fare = 279
    elif day == "Tue":
        fare = 204
    elif day == "Wed":
        fare = 204
    elif day == "Thu":
        fare = 279
    elif day == "Fri":
        fare = 370
    elif day == "Sat":
        fare = 279
    else:
        fare = 204

# Fares for Seattle-Tacoma International Airport
elif destination == "SEA":
    if day == "Mon":
        fare = 315
    elif day == "Tue":
        fare = 294
    elif day == "Wed":
        fare = 294
    elif day == "Thu":
        fare = 315
    elif day == "Fri":
        fare = 430
    elif day == "Sat":
        fare = 315
    else:
        fare = 294

# Fares for Boston Logan International Airport
elif destination == "BOS":
    if day == "Mon":
        fare = 241
    elif day == "Tue":
        fare = 205
    elif day == "Wed":
        fare = 205
    elif day == "Thu":
        fare = 241
    elif day == "Fri":
        fare = 298
    elif day == "Sat":
        fare = 241
    else:
        fare = 205



# Phase 3 & 4 | 2-1-24 11:15 PM | 2-3-23 8:23 PM

# Calculating fares if user chooses first class & additional seating options
if 1 <= row <= 7:
    first_class_fee = fare * 0.75
    fare = fare + first_class_fee

elif (row == 14) or (row == 15):
    fare = fare + 49

elif (8 <= row <= 13) or (16 <= row <= 20):
    fare = fare + 35



# Phase 5 | 2-3-23 8:35 PM

# bag_price has to be defined and evaluated if user has one bag or more
bag_price = 0

if bags == 1:
    bag_price = 35
elif bags >= 2:
    bag_price = 35 + (25 * (bags - 1))



# Phase 6

# Making adjustments to account for premium features, one free bag and free preferred seating,
# using boolean statements
if premium_plus == "y":
    fare = fare + 94

    if bags == 1:
        bag_price = 0
    else:
        bag_price = 35 + ((bags * 25) - 25)
        bag_price -= 25

    if 1 <= row <= 7:
        fare = fare - first_class_fee
    elif (row == 14) or (row == 15):
        fare = fare - 49
    elif (8 <= row <= 13) or (16 <= row <= 20):
        fare = fare - 35

# "resetting" variable values if no premium is selected
elif premium_plus == "n":
    fare = fare
    if bags >= 2:
        bag_price = 35 + (25 * (bags - 1))
    else:
        bag_price = bag_price





# calculating total price and tax last
tax = (fare * 0.18)

total = fare + tax + bag_price

# Printing and calculating ticket summary
print()
print("Ticket Price Summary")
print(f"Fare: ${fare:.2f}")

# Only printing bag price if amount of bags is greater than 0
# The same goes if premium pass is selected

if bag_price > 0:
    print(f"Bags: ${bag_price:.2f}")

if premium_plus == "y":
    print("Premium: $94.00")

print(f"Taxes: ${tax:.2f}")
print(f"Total: ${total:.2f}")
